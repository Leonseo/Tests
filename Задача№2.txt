print('На этом шаге задайте значение первого массива,используйте пробел после добавления значения')
#На этом шаге заполняем первый массив значениями
mas1 = list(map(int,input().split()))

print('На этом шаге задайте значение второго массива,используйте пробел после добавления значения')
#Заполняем второй массив значениями
mas2 = list(map(int,input().split()))

#На этом шаге определяем длину массивов
len_mas1 = len(mas1)
len_mas2 = len(mas2)

#Определяем длину результирующего массива
#Согласно правилу перемножения полиномов, количество
#коэффициентов нового полинома равна сумме количества коэффициентов
#полиномов произведения -1
len_mas3 = len_mas1 + len_mas2 - 1

#На этом шаге заполняем массив нулями
mas3 = [0 for i in range(len_mas3)]

#Проходим по элементам первого массива
for i in range(len_mas1):
    #Проходим по элементам второго массива
    for j in range(len_mas2):
        #Вычисляем значения элементов результирующего массива
        mas3[i + j] += mas1[i] * mas2[j]

#Выводим значение результирующего массива, для более эстетичного вида в
#print добавляем *
print()
print('Значение результирующего массива:')
print(*mas3)
